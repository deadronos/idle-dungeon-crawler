export const getBaseAttributes = (entityClass: EntityClass): Attributes => {
    switch (entityClass) {
        case "Warrior": return { vit: 10, str: 10, dex: 5, int: 3, wis: 3 };
        case "Cleric": return { vit: 7, str: 4, dex: 4, int: 8, wis: 10 };
